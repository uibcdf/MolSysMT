    def to_string(self, output='short_string'):

        if output=='short_string':

            string = self.name+'-'+str(self.serial)

        elif output=='long_string':

            string = self.name+'-'+str(self.serial)+'/'+self.resName+'-'+self.resSeq+'/'+self.chainId

        return string
