import molsysmt as msm
import numpy as np

molsys = msm.systems.demo['POPC']['popc.crd']
molsys = msm.convert(molsys)

axes, momenta = msm.structure.get_principal_axes(molsys)

for ii in range(3):
    print(axes[0,ii], momenta[0,ii])
